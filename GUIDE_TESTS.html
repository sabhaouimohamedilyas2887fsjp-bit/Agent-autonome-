2 Test manuel : comparer les scores 
python -c "
from scrapers.base_scraper import LegalDocument
from analyzer.criticality_scorer import CriticalityScorer

scorer = CriticalityScorer()
cas = [
    ('Bulletin Officiel', 'Dahir n 1-09-15 loi 09-08 protection des donnees cybersecurite', 'dahir'),
    ('DGSSI', 'Alerte securite : vulnerable dans les systemes', 'alerte securite'),
    ('SGG', 'Decret n 2-22-431 relatif a la procedure', 'decret'),
    ('SGG', 'Circulaire relative aux modalites de reunion', 'circulaire'),
]
print(f'{'Source':<25} {'Type':<18} {'Score':>6} {'Niveau':<10}')
print('-' * 65)
for source, title, doc_type in cas:
    doc = LegalDocument(source=source, title=title, url=f'https://test.ma/{doc_type}', doc_type=doc_type, content=title)
    r = scorer.score(doc)
    print(f'{source:<25} {doc_type:<18} {r.score:>6} {r.level:<10}')
"
✅ Résultat attendu : Dahir BO → score le plus élevé, Circulaire → score le plus bas

python -c "
from scrapers.base_scraper import LegalDocument
from analyzer.claude_analyzer import AnalysisResult
from database.db_manager import DBManager
from datetime import datetime

db = DBManager('sqlite:///veille_juridique_test.db')
doc = LegalDocument(
    source='Bulletin Officiel',
    title='TEST - Dahir relatif a la protection des donnees',
    url='https://bo.ma/test-dahir-unique-001',
    doc_type='dahir',
    content='Loi 09-08 relative a la protection des donnees personnelles.',
    published_date=datetime(2024, 1, 15),
)
orm_doc = db.upsert_document(doc)
print(f'Document insere : id={orm_doc.id}')

analysis = AnalysisResult(
    summary='Document de test pour validation du pipeline.',
    impact_level='eleve',
    tags=['test', 'donnees', 'bo'],
)
orm_an = db.save_analysis(orm_doc.id, analysis, criticality_score=72.0, model_used='test')
print(f'Analyse sauvegardee : impact={orm_an.impact_level}, score={orm_an.criticality_score}')

# Deduplication
orm_doc2 = db.upsert_document(doc)
print(f'Deduplication OK : meme id={orm_doc2.id}')

# Nettoyage
db.delete_document(orm_doc.id)
print('Document supprime.')
import os; os.remove('veille_juridique_test.db')
"
✅ Résultat attendu : insertion, analyse, déduplication et suppression sans erreur
